\begin{schema}{SetTime} % RE15

canSet : \{True, False\} \\
pressHH ::= pressed | notPressed \\
pressMM ::= pressed | notPressed \\
%pressRPT ::= Repeat | notRepeat \\

Time = pressHH \cup pressMM \\
%Repeat = pressRPT
